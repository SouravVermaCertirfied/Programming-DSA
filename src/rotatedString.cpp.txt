class Solution
{
    public:
    //Function to check if two strings are rotations of each other or not.
    bool areRotations(string s1,string s2)
    {
        // Your code here
        if(s1.length() != s2.length()){
            return false;
        }
        int len = s1.length();
        bool arr[len];
        for(int i = 0; i<len; i++){
        	arr[i] = true;
            for(int j = 0; j<len; j++){
                if(s1[j]!=s2[ (j+i)%len ]){
                    arr[i] = false;
                    break;
                }
            }
        }
        for(int i = 0; i<len; i++){
            if(arr[i]==true)
                return true;
        }
        return false;

    }
};
