class Solution
{
    public:
    //Function to find the first non-repeating character in a string.
    char nonrepeatingCharacter(string S)
    {
       //Your code here

        int arr[26];
        for(int i=0;i<26;i++){
            arr[i] = 0;
        }
       for(unsigned int i=0; i<S.length(); i++){
           arr[S[i]-97] +=1;
       }
       for(unsigned int i=0; i<S.length(); i++){
    	   int temp = (int)arr[S[i]-97];
           if(temp == 1){
               return S[i];
           }
       }
       return '$';

    }

};
