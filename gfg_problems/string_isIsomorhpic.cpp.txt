class Solution
{
    public:
    //Function to check if two strings are isomorphic.
	int k = 0;
	char v[800];
	bool insertItem(char ch1, char ch2){
		for(int i = 0;i<k;i++){
			if(ch2 == v[i*2+1]){
				return false;
			}
		}
		v[(k*2)] = ch1;
		v[(k*2+1)] = ch2;
		k++;
		return true;
	}

	int findKey(char ch1){
		for(int i = 0; i<k; i++){
			if(v[i*2] == ch1){
				return i;
			}
		}
		return -1;
	}

    bool areIsomorphic(string str1, string str2)
    {
    	int loc = 0;
    	if(str1.length()!=str2.length()){
    		return false;
    	}
    	for(unsigned int i=0; i<str1.length(); i++){
    		if(k==0){
    			insertItem(str1[i], str2[i]);
    		}
    		else{
    			loc = findKey(str1[i]);
    			if(loc == -1){
    				if(!insertItem(str1[i], str2[i])){
    					return false;
    					}
    				}
    			else{
    				if( str2[i] != v[loc*2+1]){
    					return false;
    				}
    			}
    		}
    	}
    	return true;

        // Your code here
    }
};
